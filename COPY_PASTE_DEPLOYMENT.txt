# 8. Edit .env with your OpenAI key (IMPORTANT!)
nano .env
# Press Ctrl+X, then Y, then Enter to save

# 9. Install dependencies
npm install

# 10. Build the app
npm run build

# 11. Start with PM2
pm2 start ecosystem.config.js

# 12. Save PM2 configuration
pm2 save
